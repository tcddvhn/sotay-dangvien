var SHEET_ID = "11gtYHnhty494Bl0BYs-CTV--Fg364qA082gbSSoS1kk"; // HÀM DUY NHẤT: VỪA GHI NHẬN HÀNH ĐỘNG, VỪA TRẢ VỀ SỐ LIỆU function doGet(e) { try { var sheet = SpreadsheetApp.openById(SHEET_ID).getSheets()[0]; // 1. NẾU CÓ HÀNH ĐỘNG GỬI LÊN THÌ GHI VÀO SHEET NGAY var action = e.parameter.action; var detail = e.parameter.detail || ""; // Chấp nhận thêm hành động 'chatbot' để ghi vào Sheet if (action === "visit" || action === "search" || action === "chatbot") { sheet.appendRow([new Date(), action, detail]); // Dừng lại một chút xíu để Google Sheet kịp lưu dữ liệu Utilities.sleep(500); } // 2. SAU ĐÓ, TÍNH TOÁN LẠI SỐ LIỆU MỚI NHẤT ĐỂ TRẢ VỀ CHO WEB var data = sheet.getDataRange().getValues(); var stats = { totalVisits: 0, searchCount: 0, chatbotCount: 0, // Thêm biến đếm số lượt hỏi AI dailyVisits: {} }; // Chạy vòng lặp từ dòng 2 (bỏ qua dòng tiêu đề) for (var i = 1; i < data.length; i++) { var rowDate = new Date(data[i][0]); var act = data[i][1]; if (act === "visit") { stats.totalVisits++; // Chuẩn hóa ngày tháng theo múi giờ Việt Nam (sửa lại 60000ms = 1 phút) var local = new Date(rowDate.getTime() - (rowDate.getTimezoneOffset() * 3000)); var dateStr = local.toISOString().split('T')[0]; if (!stats.dailyVisits[dateStr]) stats.dailyVisits[dateStr] = 0; stats.dailyVisits[dateStr]++; } else if (act === "search") { stats.searchCount++; } else if (act === "chatbot") { stats.chatbotCount++; // Cộng dồn số lượt hỏi AI } } // Trả kết quả về cho bảng Dashboard trên Website return ContentService.createTextOutput(JSON.stringify(stats)) .setMimeType(ContentService.MimeType.JSON); } catch(err) { return ContentService.createTextOutput(JSON.stringify({"error": err.toString()})) .setMimeType(ContentService.MimeType.JSON); } }
